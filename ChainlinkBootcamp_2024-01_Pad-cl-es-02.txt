// SPDX-License-Identifier: MIT
pragma solidity 0.8.21;
 
contract RegisterAccess {
    string[] private info;
    address public owner;
    mapping (address => bool) public allowlist; 
 
    constructor() {
        owner = msg.sender;
        allowlist[msg.sender] = true;
    } 
 
    event InfoChange(string oldInfo, string newInfo); 
 
    modifier onlyOwner {
        require(msg.sender == owner,"Only owner");
        _;
    }
 
    modifier onlyAllowlist {
        require(allowlist[msg.sender] == true, "Only allowlist");
        _;
    } 
 
    function getInfo(uint index) public view returns (string memory) {
        return info[index];
    }
 
    function setInfo(uint index, string memory _info) public onlyAllowlist {
        emit InfoChange (info[index], _info);
        info[index] = _info;
    }
 
    function addInfo(string memory _info) public onlyAllowlist returns (uint index) {
        info.push (_info);
        index = info.length -1;
    }
   
    function listInfo() public view returns (string[] memory) {
        return info;
    }
 
    function addMember (address _member) public onlyOwner {        
        allowlist[_member] = true;
    }
 
    function delMember (address _member) public onlyOwner {
        allowlist[_member] = false;
    }    
}


// Fin

Este es un contrato inteligente de Ethereum escrito en Solidity. El contrato se llama `RegisterAccess` y su propósito principal es gestionar una lista de información (`info`) y controlar quién puede acceder y modificar esa lista.

Aquí está lo que hace cada parte:

- `string[] private info;`: Declara una variable de estado privada llamada `info` que es un array de strings.

- `address public owner;`: Declara una variable de estado pública llamada `owner` que almacena una dirección de Ethereum.

- `mapping (address => bool) public allowlist;`: Declara una variable de estado pública llamada `allowlist` que es un mapeo de direcciones de Ethereum a booleanos. Este mapeo se utiliza para controlar qué direcciones tienen permiso para modificar la lista de `info`.

- `constructor() { ... }`: Este es el constructor del contrato, que se ejecuta una vez cuando se despliega el contrato. Establece al propietario del contrato como el remitente del mensaje (`msg.sender`) y añade al remitente a la lista de permitidos.

- `event InfoChange(string oldInfo, string newInfo);`: Declara un evento que se puede emitir cuando la información cambia. Los eventos son una forma de registrar la actividad en la blockchain.

- `modifier onlyOwner { ... }`: Declara un modificador que requiere que el remitente del mensaje sea el propietario del contrato. Este modificador se puede añadir a las funciones para restringir su uso al propietario del contrato.

- `modifier onlyAllowlist { ... }`: Declara un modificador que requiere que el remitente del mensaje esté en la lista de permitidos. Este modificador se puede añadir a las funciones para restringir su uso a las direcciones en la lista de permitidos.

- `function getInfo(uint index) public view returns (string memory) { ... }`: Esta función devuelve el elemento de `info` en el índice especificado.

- `function setInfo(uint index, string memory _info) public onlyAllowlist { ... }`: Esta función establece el elemento de `info` en el índice especificado. Emite un evento `InfoChange` y sólo puede ser llamada por direcciones en la lista de permitidos.

- `function addInfo(string memory _info) public onlyAllowlist returns (uint index) { ... }`: Esta función añade un nuevo elemento a `info` y devuelve su índice. Sólo puede ser llamada por direcciones en la lista de permitidos.

- `function listInfo() public view returns (string[] memory) { ... }`: Esta función devuelve toda la lista de `info`.

- `function addMember (address _member) public onlyOwner { ... }`: Esta función añade una dirección a la lista de permitidos. Sólo puede ser llamada por el propietario del contrato.

- `function delMember (address _member) public onlyOwner { ... }`: Esta función elimina una dirección de la lista de permitidos. Sólo puede ser llamada por el propietario del contrato.

En resumen, este contrato permite al propietario del contrato gestionar una lista de direcciones permitidas. Las direcciones en la lista de permitidos pueden ver y modificar una lista de información.
